class CausalAttention(nn.Module):

    def __init__(self, d_in, d_out, block_size, dropout, qkv_bias=False):
        '''
        ÊûÑÈÄ†ÂáΩÊï∞ÔºåËæìÂÖ•ÂèÇÊï∞Â¶Ç‰∏ãÔºö
        d_in: ËæìÂÖ•ÁöÑÁª¥Â∫¶
        d_out: ËæìÂá∫ÁöÑÁª¥Â∫¶
        block_size: Ê≥®ÊÑèÂäõÊùÉÈáçÁü©ÈòµÁöÑÂ§ßÂ∞è
        dropout: dropout ÊØî‰æã
        qkv_bias: ÊòØÂê¶ÂØπ query„ÄÅkey Âíå value Âä†ÂÅèÁΩÆ
        '''
        super().__init__()
        self.d_out = d_out
        # Ê†πÊçÆÂâçÊñáÔºåÊØè‰∏Ä‰∏™ÊùÉÈáçÁü©ÈòµÈÉΩÊòØ d_in x d_out ÁöÑÁ∫øÊÄßÂ±Ç
        self.W_query = nn.Linear(d_in, d_out, bias=qkv_bias)
        self.W_key   = nn.Linear(d_in, d_out, bias=qkv_bias)
        self.W_value = nn.Linear(d_in, d_out, bias=qkv_bias)
        # ‰∏Ä‰∏™ dropout Â±Ç
        self.dropout = nn.Dropout(dropout) 
        # ‰∏Ä‰∏™Êé©Á†ÅÁü©ÈòµÔºå‰∏ã‰∏âËßí‰∏∫ 1ÔºåÂÖ∂‰Ωô‰∏∫ 0
        self.register_buffer('mask', torch.triu(torch.ones(block_size, block_size), diagonal=1)) # New

    def forward(self, x):
        '''
        ÂâçÂêë‰º†Êí≠ÂáΩÊï∞ÔºåËæìÂÖ•ÂèÇÊï∞‰∏∫ xÔºåÁª¥Â∫¶‰∏∫ b x num_tokens x d_inÔºåËæìÂá∫Áª¥Â∫¶‰∏∫ b x num_tokens x d_out
        '''
        b, num_tokens, d_in = x.shape
        keys = self.W_key(x)
        queries = self.W_query(x)
        values = self.W_value(x)
        # transpose ÊòØ‰∏∫‰∫ÜÂÆûÁé∞Áü©Èòµ‰πòÊ≥ï
        attn_scores = queries @ keys.transpose(1, 2)
        # Âç≥‰∏äÊñáËØ¥ËøáÁöÑÔºåÂ∞ÜÊé©Á†Å‰ªé 0 ‰øÆÊîπ‰∏∫ -infÔºåÂÜçËøõË°åÈÅÆËîΩÊìç‰Ωú
        attn_scores.masked_fill_(  # New, _ ops are in-place
            self.mask.bool()[:num_tokens, :num_tokens], -torch.inf)
        # ÁªèËøá softmax 
        attn_weights = torch.softmax(attn_scores / keys.shape[-1]**0.5, dim=1)
        # ËøõË°å dropout
        attn_weights = self.dropout(attn_weights) # New
        # ÂæóÂà∞ÊúÄÂêéÁªìÊûú
        context_vec = attn_weights @ values
        return context_vec
